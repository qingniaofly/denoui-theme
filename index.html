denoui-theme-bg

denoui-theme-bg-008

denoui-theme-border

denoui-theme-test

使用：

引入js，添加如下代码

            
    var options = {
        id: "denoui",
        color: "rgba(242,143,0,1)",
        data:{
            "theme-test": {"color":true}
        },
    }
    denouiTheme.render(options)

denouiTheme

方法名 参数 用途 返回值 例子
get (id) 获取指定id的 theme style Style实例 
                        
    denouiTheme.get("id")
render (options) 渲染 theme style 无 
                        
    var options = {
        id: "denoui",
        color: "rgba(242,143,0,1)",
        data:{
            "theme-test": {"color":true}
        },
    }
    denouiTheme.render(options)
rgbaToHexify (rgba) rgba转16进制 16进制字符串 
                        
    denouiTheme.rgbaToHexify('rgba(242,143,0,1)') // = > "#f28f00"

参数说明-options

参数名 数据类型 说明 默认值
options object - 
            
    {
        id: "denoui",
        color: "rgb(47,82,635)",
        data:{
            "theme-text": { "color": true },
            "theme-bg": { "background-color": true },
            "theme-bg-008": { "background-color": true, "opacity": 0.08 },
            "theme-border": { "border-color": true }
        }
    }
id string - denoui
color string 必须是正确的rgb/rgba值 rgb(47,82,635)
data object {} 
            
        {
            "theme-text": { "color": true },
            "theme-bg": { "background-color": true },
            "theme-bg-008": { "background-color": true, "opacity": 0.08 },
            "theme-border": { "border-color": true }
        }

Style实例

            
    var style = denouiTheme.get("denoui") // => Style实例
方法名 参数 用途 返回值 例子
updateColor (rgb) 更新主题色 当前实例 
                        
    denouiStyle.updateColor(rgb)
load (opt) 加载指定主题配置 当前实例 
                        
    denouiStyle.load(opt)
updateStyle (name,opt) 更新指定类主题 当前实例 
                        
    denouiStyle.updateStyle('theme-bg-008',{ opacity: 0.2 })
